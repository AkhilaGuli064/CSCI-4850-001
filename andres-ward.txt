CREATE TABLE `airplane` (
  `airplane_id` int(11) NOT NULL,
  `make` varchar(20) NOT NULL,
  `model` varchar(30) NOT NULL,
  `factory_date` date NOT NULL,
  `miles` int(15) NOT NULL,
  `passenger_capacity` int(4) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `airport` (
  `airport_id` char(3) NOT NULL,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `arrival` (
  `flight_id` int(11) NOT NULL,
  `airport_id` char(3) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `attendants_scheduled` (
  `crew_manifest_id` int(11) NOT NULL,
  `attendant_id1` int(11) NOT NULL,
  `attendant_id2` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `copilot_scheduled` (
  `crew_manifest_id` int(11) NOT NULL,
  `copilot_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `crew_manifest` (
  `crew_manifest_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `crew_scheduled` (
  `flight_id` int(11) NOT NULL,
  `crew_manifest_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `customer` (
  `customer_id` int(11) NOT NULL,
  `street_number` varchar(10) NOT NULL,
  `street_name` varchar(30) NOT NULL,
  `apt_number` varchar(10) DEFAULT NULL,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL,
  `zip` char(5) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `departure` (
  `flight_id` int(11) NOT NULL,
  `airport_id` char(3) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `employee` (
  `employee_id` int(11) NOT NULL,
  `birth_date` date NOT NULL,
  `gender` char(1) NOT NULL,
  `street_number` varchar(10) NOT NULL,
  `street_name` varchar(30) NOT NULL,
  `apt_number` varchar(10) DEFAULT NULL,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL,
  `zip` char(5) NOT NULL,
  `start_date` date NOT NULL,
  `salary` decimal(9,2) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `flight` (
  `flight_id` int(11) NOT NULL,
  `departure_date` date NOT NULL,
  `departure_time` time NOT NULL,
  `arrival_date` date NOT NULL,
  `arrival_time` time NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `flight_attendant` (
  `attendant_id` int(11) NOT NULL,
  `flight_hours` decimal(12,2) NOT NULL,
  `cell_phone` char(10) NOT NULL,
  `rank` varchar(15) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `has` (
  `ticket_id` int(11) NOT NULL,
  `passenger_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `home_airport` (
  `employee_id` int(11) NOT NULL,
  `airport_id` char(3) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `lead_scheduled` (
  `crew_manifest_id` int(11) NOT NULL,
  `lead_attendant_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `passenger` (
  `passenger_id` int(11) NOT NULL,
  `birth_date` date NOT NULL,
  `gender` char(1) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `person` (
  `person_id` int(11) NOT NULL,
  `first_name` varchar(30) NOT NULL,
  `last_name` varchar(30) NOT NULL,
  `phone_number` char(10) NOT NULL,
  `email` varchar(30) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `pilot` (
  `pilot_id` int(11) NOT NULL,
  `pilot_hours` decimal(12,2) NOT NULL,
  `cell_phone` char(10) NOT NULL,
  `rank` varchar(15) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `plane_scheduled` (
  `flight_id` int(11) NOT NULL,
  `airplane_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `purchases` (
  `ticket_id` int(11) NOT NULL,
  `customer_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `seats_filled` (
  `ticket_id` int(11) NOT NULL,
  `flight_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `senior_scheduled` (
  `crew_manifest_id` int(11) NOT NULL,
  `senior_pilot_id` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `ticket` (
  `ticket_id` int(11) NOT NULL,
  `seat_number` int(11) NOT NULL,
  `price` decimal(7,2) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

4. Original SQL specification from Phase 1

CREATE TABLE `airplane` (
  `airplane_id` int(11) NOT NULL PRIMARY KEY,
  `make` varchar(20) NOT NULL,
  `model` varchar(30) NOT NULL,
  `factory_date` date NOT NULL,
  `miles` int(15) NOT NULL,
  `passenger_capacity` int(4) NOT NULL
) 

CREATE TABLE `airport` (
  `airport_id` char(3) NOT NULL PRIMARY KEY,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL
) 

CREATE TABLE `arrival` (
  `flight_id` int(11) NOT NULL PRIMARY KEY,
  `airport_id` char(3) NOT NULL,
  FOREIGN KEY (`flight_id`) REFERENCES `flight` (`flight_id`),
  FOREIGN KEY (`airport_id`) REFERENCES `airport` (`airport_id`)
) 

CREATE TABLE `attendants_scheduled` (
  `crew_manifest_id` int(11) NOT NULL PRIMARY KEY,
  `attendant_id1` int(11) NOT NULL,
  `attendant_id2` int(11) NOT NULL,
  FOREIGN KEY (`crew_manifest_id`) REFERENCES `crew_manifest` (`crew_manifest_id`),
  FOREIGN KEY (`attendant_id1`) REFERENCES `flight_attendant` (`attendant_id`),
  FOREIGN KEY (`attendant_id2`) REFERENCES `flight_attendant` (`attendant_id`)
) 

CREATE TABLE `copilot_scheduled` (
  `crew_manifest_id` int(11) NOT NULL PRIMARY KEY,
  `copilot_id` int(11) NOT NULL,
  FOREIGN KEY (`crew_manifest_id`) REFERENCES `crew_manifest` (`crew_manifest_id`),
  FOREIGN KEY (`copilot_id`) REFERENCES `pilot` (`pilot_id`)
) 

CREATE TABLE `crew_manifest` (
  `crew_manifest_id` int(11) NOT NULL PRIMARY KEY
) 

CREATE TABLE `crew_scheduled` (
  `flight_id` int(11) NOT NULL PRIMARY KEY,
  `crew_manifest_id` int(11) NOT NULL,
  FOREIGN KEY (`flight_id`) REFERENCES `flight` (`flight_id`),
  FOREIGN KEY (`crew_manifest_id`) REFERENCES `crew_manifest` (`crew_manifest_id`)
) 

CREATE TABLE `customer` (
  `customer_id` int(11) NOT NULL PRIMARY KEY,
  `street_number` varchar(10) NOT NULL,
  `street_name` varchar(30) NOT NULL,
  `apt_number` varchar(10) DEFAULT NULL,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL,
  `zip` char(5) NOT NULL,
  FOREIGN KEY (`customer_id`) REFERENCES `person` (`person_id`)
) 

CREATE TABLE `departure` (
  `flight_id` int(11) NOT NULL PRIMARY KEY,
  `airport_id` char(3) NOT NULL,
  FOREIGN KEY (`flight_id`) REFERENCES `flight` (`flight_id`),
  FOREIGN KEY (`airport_id`) REFERENCES `airport` (`airport_id`)
) 

CREATE TABLE `employee` (
  `employee_id` int(11) NOT NULL PRIMARY KEY,
  `birth_date` date NOT NULL,
  `gender` char(1) NOT NULL,
  `street_number` varchar(10) NOT NULL,
  `street_name` varchar(30) NOT NULL,
  `apt_number` varchar(10) DEFAULT NULL,
  `city` varchar(30) NOT NULL,
  `state` char(2) NOT NULL,
  `zip` char(5) NOT NULL,
  `start_date` date NOT NULL,
  `salary` decimal(9,2) NOT NULL,
  FOREIGN KEY (`employee_id`) REFERENCES `person` (`person_id`)
) 

CREATE TABLE `flight` (
  `flight_id` int(11) NOT NULL PRIMARY KEY,
  `departure_date` date NOT NULL,
  `departure_time` time NOT NULL,
  `arrival_date` date NOT NULL,
  `arrival_time` time NOT NULL
) 

CREATE TABLE `flight_attendant` (
  `attendant_id` int(11) NOT NULL PRIMARY KEY,
  `flight_hours` decimal(12,2) NOT NULL,
  `cell_phone` char(10) NOT NULL,
  `rank` varchar(15) NOT NULL,
  FOREIGN KEY (`attendant_id`) REFERENCES `employee` (`employee_id`)
) 

CREATE TABLE `has` (
  `ticket_id` int(11) NOT NULL PRIMARY KEY,
  `passenger_id` int(11) NOT NULL,
  FOREIGN KEY (`ticket_id`) REFERENCES `ticket` (`ticket_id`),
  FOREIGN KEY (`passenger_id`) REFERENCES `passenger` (`passenger_id`)
) 

CREATE TABLE `home_airport` (
  `employee_id` int(11) NOT NULL PRIMARY KEY,
  `airport_id` char(3) NOT NULL,
  FOREIGN KEY (`employee_id`) REFERENCES `employee` (`employee_id`),
  FOREIGN KEY (`airport_id`) REFERENCES `airport` (`airport_id`)
) 

CREATE TABLE `lead_scheduled` (
  `crew_manifest_id` int(11) NOT NULL PRIMARY KEY,
  `lead_attendant_id` int(11) NOT NULL,
  FOREIGN KEY (`crew_manifest_id`) REFERENCES `crew_manifest` (`crew_manifest_id`),
  FOREIGN KEY (`lead_attendant_id`) REFERENCES `flight_attendant` (`attendant_id`)
) 

CREATE TABLE `passenger` (
  `passenger_id` int(11) NOT NULL PRIMARY KEY,
  `birth_date` date NOT NULL,
  `gender` char(1) NOT NULL,
  FOREIGN KEY (`passenger_id`) REFERENCES `person` (`person_id`)
) 

CREATE TABLE `person` (
  `person_id` int(11) NOT NULL PRIMARY KEY,
  `first_name` varchar(30) NOT NULL,
  `last_name` varchar(30) NOT NULL,
  `phone_number` char(10) NOT NULL,
  `email` varchar(30) NOT NULL
) 

CREATE TABLE `pilot` (
  `pilot_id` int(11) NOT NULL PRIMARY KEY,
  `pilot_hours` decimal(12,2) NOT NULL,
  `cell_phone` char(10) NOT NULL,
  `rank` varchar(15) NOT NULL,
  FOREIGN KEY (`pilot_id`) REFERENCES `employee` (`employee_id`)
) 

CREATE TABLE `plane_scheduled` (
  `flight_id` int(11) NOT NULL PRIMARY KEY,
  `airplane_id` int(11) NOT NULL,
  FOREIGN KEY (`flight_id`) REFERENCES `flight` (`flight_id`),
  FOREIGN KEY (`airplane_id`) REFERENCES `airplane` (`airplane_id`)
) 

CREATE TABLE `purchases` (
  `ticket_id` int(11) NOT NULL PRIMARY KEY,
  `customer_id` int(11) NOT NULL,
  FOREIGN KEY (`ticket_id`) REFERENCES `ticket` (`ticket_id`),
  FOREIGN KEY (`customer_id`) REFERENCES `customer` (`customer_id`)
) 

CREATE TABLE `seats_filled` (
  `ticket_id` int(11) NOT NULL PRIMARY KEY,
  `flight_id` int(11) NOT NULL,
  FOREIGN KEY (`ticket_id`) REFERENCES `ticket` (`ticket_id`),
  FOREIGN KEY (`flight_id`) REFERENCES `flight` (`flight_id`)
) 

CREATE TABLE `senior_scheduled` (
  `crew_manifest_id` int(11) NOT NULL PRIMARY KEY,
  `senior_pilot_id` int(11) NOT NULL,
  FOREIGN KEY (`crew_manifest_id`) REFERENCES `crew_manifest` (`crew_manifest_id`),
  FOREIGN KEY (`senior_pilot_id`) REFERENCES `pilot` (`pilot_id`)
) 

CREATE TABLE `ticket` (
  `ticket_id` int(11) NOT NULL PRIMARY KEY,
  `seat_number` int(11) NOT NULL,
  `price` decimal(7,2) NOT NULL
)
